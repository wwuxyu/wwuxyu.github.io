ListNode* swap(ListNode* head) {
	ListNode* nhead = new ListNode(-1);
  nhead->next = head;
  
  ListNode* tmp = nhead;
  while (tmp->next != nullptr && tmp->next->next != nullptr) {
    ListNode* p1 = tmp->next;
    ListNode* p2 = tmp->next->next;
    tmp->next = p2;
    p1->next = p2->next;
    p2->next = p1;
    tmp = p1;
  }
  return nhead->next;
}

class Solution {
public:
    int ladderLength(string beginWord, string endWord, vector<string>& wordList) {
        unordered_set<string> dict(wordList.begin(), wordList.end());        
        if (!dict.count(endWord)) return 0;
        
        queue<string> q;
        q.push(beginWord);
        
        int l = beginWord.length();
        int step = 0;
        
        while (!q.empty()) {
            ++step;
            for (int size = q.size(); size > 0; size--) {                
                string w = q.front();                
                q.pop();
                for (int i = 0; i < l; i++) {                
                    char ch = w[i];
                    for (int j = 'a'; j <= 'z'; j++) {
                        w[i] = j;
                        // Found the solution
                        if (w == endWord) return step + 1;
                        // Not in dict, skip it
                        if (!dict.count(w)) continue;
                        // Remove new word from dict
                        dict.erase(w);
                        // Add new word into queue
                        q.push(w);                    
                    }
                    w[i] = ch;
                }
            }
        }
        return 0;
    }
};

class Solution {
public:
    vector<vector<string>> findLadders(string beginWord, string endWord, vector<string>& wordList) {
        vector<vector<string>> ans;
        unordered_set<string> dict(wordList.begin(), wordList.end());
        if (!dict.count(endWord)) return ans;
        dict.erase(endWord);
        dict.erase(beginWord);

        unordered_map<string, int> steps{{beginWord, 1}};
        unordered_map<string, vector<string>> parents;
        queue<string> q;
        q.push(beginWord);

        const int len = beginWord.size();
        int step = 0;
        bool found = false;
        while (!q.empty() && !found) {
            ++step;
            for (int size=q.size();size>0;size--) {
                const string p = q.front(); q.pop();
                string cur = p;
                for (int i=0;i<len;i++) {
                    const char ch = cur[i];
                    for (int j='a';j<='z';j++) {
                        if (j == ch) continue;
                        cur[i] = j;
                        if (cur == endWord) {
                            parents[cur].push_back(p);                         
                            found = true;
                        }
                        else {
                            // 同一层的step
                            if (steps.count(cur) && step +1 == steps.at(cur)) {
                                parents[cur].push_back(p);
                            }
                        }

                        // 如果该单词不在字典里，即跳过，若在则从字典中删除。
                        if (!dict.count(cur)) continue;
                        dict.erase(cur);
                        q.push(cur);
                        steps[cur] = steps.at(p) + 1;
                        parents[cur].push_back(p);
                    }
                    cur[i] = ch;
                }
            }
        }

        if (found) {
            vector<string> path{endWord};
            dfs(endWord, beginWord, parents, path, ans);
        }
        return ans;
    }

    void dfs(const string& word,
             const string& beginWord,
             const unordered_map<string, vector<string>>& parents,
             vector<string>& path,
             vector<vector<string>>& ans) {
                if (word == beginWord) {
                    ans.push_back(vector<string>(path.rbegin(), path.rend()));
                    return;
                }
                for (const auto& it : parents.at(word)) {
                    path.push_back(it);
                    dfs(it, beginWord, parents, path, ans);
                    path.pop_back();
                }
            }
};
